-- =========================================
-- БАЗА ДАННЫХ (PostgreSQL)
-- =========================================

-- ====== АДРЕСА ======

CREATE TABLE cities (
    id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);

CREATE TABLE streets (
    id INT PRIMARY KEY,
    city_id INT NOT NULL,
    name VARCHAR(150) NOT NULL,
    CONSTRAINT fk_street_city
        FOREIGN KEY (city_id)
        REFERENCES cities(id)
        ON DELETE CASCADE
);

CREATE TABLE houses (
    id INT PRIMARY KEY,
    street_id INT NOT NULL,
    house_number VARCHAR(20) NOT NULL,
    CONSTRAINT fk_house_street
        FOREIGN KEY (street_id)
        REFERENCES streets(id)
        ON DELETE CASCADE
);

-- ====== СОТРУДНИКИ ======

CREATE TABLE users (
    id INT PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(100) NOT NULL,
    role VARCHAR(20) NOT NULL
        CHECK (role IN ('admin', 'manager', 'agent')),
    phone VARCHAR(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ====== КЛИЕНТЫ (БЕЗ ПАСПОРТНЫХ ДАННЫХ) ======

CREATE TABLE clients (
    id INT PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    phone_primary VARCHAR(20),
    email VARCHAR(255),
    birth_date DATE
);

-- ====== ОБЪЕКТЫ НЕДВИЖИМОСТИ ======

CREATE TABLE properties (
    id INT PRIMARY KEY,
    title VARCHAR(255),

    operation_type VARCHAR(10) NOT NULL
        CHECK (operation_type IN ('sale', 'rent')),

    city_id INT NOT NULL,
    street_id INT NOT NULL,
    house_id INT NOT NULL,
    apartment_number VARCHAR(20),

    coordinates_lat DECIMAL(10,8),
    coordinates_lon DECIMAL(11,8),

    price FLOAT NOT NULL,
    area_total FLOAT,
    area_living FLOAT,

    status VARCHAR(10) DEFAULT 'active'
        CHECK (status IN ('active', 'reserved', 'sold', 'rented')),

    CONSTRAINT fk_property_city FOREIGN KEY (city_id) REFERENCES cities(id),
    CONSTRAINT fk_property_street FOREIGN KEY (street_id) REFERENCES streets(id),
    CONSTRAINT fk_property_house FOREIGN KEY (house_id) REFERENCES houses(id)
);

-- ====== ФОТО ОБЪЕКТОВ ======

CREATE TABLE property_photos (
    id INT PRIMARY KEY,
    property_id INT NOT NULL,
    url TEXT NOT NULL,
    CONSTRAINT fk_photo_property
        FOREIGN KEY (property_id)
        REFERENCES properties(id)
        ON DELETE CASCADE
);

-- ====== ДОКУМЕНТЫ ОБЪЕКТОВ ======

CREATE TABLE property_documents (
    id INT PRIMARY KEY,
    property_id INT NOT NULL,
    file_url TEXT NOT NULL,
    is_verified BOOLEAN DEFAULT FALSE,
    uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_document_property
        FOREIGN KEY (property_id)
        REFERENCES properties(id)
        ON DELETE CASCADE
);

-- ====== ЗАЯВКИ ======

CREATE TABLE requests (
    id INT PRIMARY KEY,
    client_id INT NOT NULL,
    agent_id INT NOT NULL,

    operation_type VARCHAR(10) NOT NULL
        CHECK (operation_type IN ('sale', 'rent')),

    property_type VARCHAR(50),

    status VARCHAR(15) DEFAULT 'open'
        CHECK (status IN ('open', 'processing', 'closed')),

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_request_client FOREIGN KEY (client_id) REFERENCES clients(id),
    CONSTRAINT fk_request_agent FOREIGN KEY (agent_id) REFERENCES users(id)
);

-- ====== СДЕЛКИ ======

CREATE TABLE deals (
    id SERIAL PRIMARY KEY,
    deal_number VARCHAR(50) NOT NULL,
    property_id INT NOT NULL,
    agent_id INT NOT NULL,

    type VARCHAR(10) NOT NULL
        CHECK (type IN ('sale', 'rent')),

    price_object NUMERIC(12,2) NOT NULL,
    commission_percent NUMERIC(5,2),
    commission_amount NUMERIC(12,2),
    deal_date DATE,

    CONSTRAINT fk_deal_property FOREIGN KEY (property_id) REFERENCES properties(id),
    CONSTRAINT fk_deal_agent FOREIGN KEY (agent_id) REFERENCES users(id)
);
