-- =========================================
-- ЕДИНАЯ СХЕМА БД (PostgreSQL)
-- =========================================

CREATE TABLE operation_types (
    id INT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(10) UNIQUE NOT NULL,
    name VARCHAR(50) NOT NULL
);

CREATE TABLE property_statuses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(10) UNIQUE NOT NULL,
    name VARCHAR(50) NOT NULL
);

CREATE TABLE request_statuses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(15) UNIQUE NOT NULL,
    name VARCHAR(50) NOT NULL
);

CREATE TABLE deal_statuses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(20) UNIQUE NOT NULL,
    name VARCHAR(50) NOT NULL,
    `order` SMALLINT UNSIGNED DEFAULT 0
);

CREATE TABLE task_statuses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(20) UNIQUE NOT NULL,
    name VARCHAR(50) NOT NULL,
    `order` SMALLINT UNSIGNED DEFAULT 0
);

CREATE TABLE user_roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    code VARCHAR(20) UNIQUE NOT NULL,
    name VARCHAR(50) NOT NULL,
    description TEXT,
    max_active_tasks SMALLINT UNSIGNED DEFAULT 2,
    max_in_progress_tasks SMALLINT UNSIGNED DEFAULT 1,
    max_active_requests SMALLINT UNSIGNED DEFAULT 2
);

CREATE TABLE cities (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    region VARCHAR(100),
    external_id CHAR(36)
);

CREATE TABLE streets (
    id INT AUTO_INCREMENT PRIMARY KEY,
    city_id INT NOT NULL,
    name VARCHAR(150) NOT NULL,
    street_type VARCHAR(20),
    external_id CHAR(36),
    FOREIGN KEY (city_id) REFERENCES cities(id)
);

CREATE TABLE houses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    street_id INT NOT NULL,
    house_number VARCHAR(20) NOT NULL,
    building VARCHAR(10),
    structure VARCHAR(10),
    postal_code VARCHAR(10),
    external_id CHAR(36),
    FOREIGN KEY (street_id) REFERENCES streets(id)
);

CREATE TABLE addresses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    house_id INT NOT NULL,
    apartment_number VARCHAR(20),
    entrance INT,
    floor INT,
    CONSTRAINT unique_address UNIQUE (house_id, apartment_number),
    FOREIGN KEY (house_id) REFERENCES houses(id)
);

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    phone VARCHAR(20) UNIQUE,
    user_type VARCHAR(20) DEFAULT 'client',
    role_id INT,
    password VARCHAR(255) NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    is_staff BOOLEAN DEFAULT FALSE,
    is_email_verified BOOLEAN DEFAULT FALSE,
    is_phone_verified BOOLEAN DEFAULT FALSE,
    last_login DATETIME,
    last_ip VARCHAR(45),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (role_id) REFERENCES user_roles(id)
);

CREATE TABLE employee_profiles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT UNIQUE NOT NULL,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    middle_name VARCHAR(50),
    position VARCHAR(100),
    department VARCHAR(100),
    hire_date DATE,
    internal_phone VARCHAR(20),
    notes TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE client_profiles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT UNIQUE NOT NULL,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    middle_name VARCHAR(50),
    birth_date DATE,
    passport_series VARCHAR(4),
    passport_number VARCHAR(6),
    passport_issued_by VARCHAR(255),
    passport_issued_date DATE,
    passport_code VARCHAR(7),
    registration_address TEXT,
    actual_address TEXT,
    preferred_contact_method VARCHAR(20),
    notes TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE properties (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255),
    operation_type_id INT NOT NULL,
    status_id INT NOT NULL,
    address_id INT NOT NULL,
    coordinates_lat DECIMAL(10,8),
    coordinates_lon DECIMAL(11,8),
    price FLOAT NOT NULL,
    price_per_sqm FLOAT,
    area_total DECIMAL(8,2),
    area_living DECIMAL(8,2),
    area_kitchen DECIMAL(8,2),
    rooms_count INT,
    floor_number INT,
    total_floors INT,
    description TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (operation_type_id) REFERENCES operation_types(id),
    FOREIGN KEY (status_id) REFERENCES property_statuses(id),
    FOREIGN KEY (address_id) REFERENCES addresses(id)
);

CREATE TABLE property_features (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    category VARCHAR(50)
);

CREATE TABLE property_feature_values (
    id INT AUTO_INCREMENT PRIMARY KEY,
    property_id INT NOT NULL,
    feature_id INT NOT NULL,
    value VARCHAR(255),
    CONSTRAINT unique_property_feature UNIQUE (property_id, feature_id),
    FOREIGN KEY (property_id) REFERENCES properties(id),
    FOREIGN KEY (feature_id) REFERENCES property_features(id)
);

CREATE TABLE property_photos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    property_id INT NOT NULL,
    image VARCHAR(255),
    url TEXT,
    caption VARCHAR(255),
    is_cover BOOLEAN DEFAULT FALSE,
    is_hidden BOOLEAN DEFAULT FALSE,
    `order` INT UNSIGNED DEFAULT 0,
    uploaded_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (property_id) REFERENCES properties(id)
);

CREATE TABLE property_documents (
    id INT AUTO_INCREMENT PRIMARY KEY,
    property_id INT NOT NULL,
    document_name VARCHAR(255) NOT NULL,
    file_url TEXT NOT NULL,
    is_verified BOOLEAN DEFAULT FALSE,
    verified_by_id INT,
    verified_at DATETIME,
    uploaded_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (property_id) REFERENCES properties(id),
    FOREIGN KEY (verified_by_id) REFERENCES users(id)
);

CREATE TABLE requests (
    id INT AUTO_INCREMENT PRIMARY KEY,
    client_id INT NOT NULL,
    agent_id INT,
    property_id INT,
    operation_type_id INT NOT NULL,
    status_id INT NOT NULL,
    property_type VARCHAR(50),
    min_price FLOAT,
    max_price FLOAT,
    min_area DECIMAL(8,2),
    max_area DECIMAL(8,2),
    rooms_count INT,
    address_preferences TEXT,
    description TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    closed_at DATETIME,
    FOREIGN KEY (client_id) REFERENCES users(id),
    FOREIGN KEY (agent_id) REFERENCES users(id),
    FOREIGN KEY (property_id) REFERENCES properties(id),
    FOREIGN KEY (operation_type_id) REFERENCES operation_types(id),
    FOREIGN KEY (status_id) REFERENCES request_statuses(id)
);

CREATE TABLE deals (
    id INT AUTO_INCREMENT PRIMARY KEY,
    deal_number VARCHAR(50) UNIQUE NOT NULL,
    property_id INT NOT NULL,
    agent_id INT NOT NULL,
    client_id INT NOT NULL,
    operation_type_id INT NOT NULL,
    status_id INT,
    request_id INT UNIQUE,
    price_final FLOAT NOT NULL,
    commission_percent DECIMAL(5,2),
    commission_amount FLOAT,
    deal_date DATE NOT NULL,
    notes TEXT,
    contract_file VARCHAR(255),
    contract_status VARCHAR(20) DEFAULT 'not_requested',
    contract_error_message TEXT,
    contract_requested_at DATETIME,
    contract_processing_started_at DATETIME,
    contract_generated_at DATETIME,
    FOREIGN KEY (property_id) REFERENCES properties(id),
    FOREIGN KEY (agent_id) REFERENCES users(id),
    FOREIGN KEY (client_id) REFERENCES users(id),
    FOREIGN KEY (operation_type_id) REFERENCES operation_types(id),
    FOREIGN KEY (status_id) REFERENCES deal_statuses(id),
    FOREIGN KEY (request_id) REFERENCES requests(id)
);

CREATE TABLE tasks (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    priority VARCHAR(10) DEFAULT 'normal',
    task_type VARCHAR(30) DEFAULT 'other',
    status_id INT NOT NULL,
    assignee_id INT NOT NULL,
    created_by_id INT NOT NULL,
    client_id INT,
    property_id INT,
    request_id INT,
    deal_id INT,
    due_date DATETIME,
    completed_at DATETIME,
    result TEXT,
    steps_log JSON,
    is_auto_closed BOOLEAN DEFAULT FALSE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (status_id) REFERENCES task_statuses(id),
    FOREIGN KEY (assignee_id) REFERENCES users(id),
    FOREIGN KEY (created_by_id) REFERENCES users(id),
    FOREIGN KEY (client_id) REFERENCES users(id),
    FOREIGN KEY (property_id) REFERENCES properties(id),
    FOREIGN KEY (request_id) REFERENCES requests(id),
    FOREIGN KEY (deal_id) REFERENCES deals(id)
);
