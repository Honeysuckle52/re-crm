-- ============================================
-- 1. СПРАВОЧНИКИ (Lookup Tables)
-- ============================================

CREATE TABLE `operation_types` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(10) NOT NULL UNIQUE,
    `name` VARCHAR(50) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `property_statuses` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(10) NOT NULL UNIQUE,
    `name` VARCHAR(50) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `request_statuses` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(15) NOT NULL UNIQUE,
    `name` VARCHAR(50) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `deal_statuses` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(20) NOT NULL UNIQUE,
    `name` VARCHAR(50) NOT NULL,
    `order` SMALLINT UNSIGNED NOT NULL DEFAULT 0,
    PRIMARY KEY (`id`)
);

CREATE TABLE `task_statuses` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(20) NOT NULL UNIQUE,
    `name` VARCHAR(50) NOT NULL,
    `order` SMALLINT UNSIGNED NOT NULL DEFAULT 0,
    PRIMARY KEY (`id`)
);

CREATE TABLE `property_types` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(50) NOT NULL UNIQUE,
    `name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `task_priorities` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(50) NOT NULL UNIQUE,
    `name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `task_types` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(50) NOT NULL UNIQUE,
    `name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `client_kinds` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(50) NOT NULL UNIQUE,
    `name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `contact_methods` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(50) NOT NULL UNIQUE,
    `name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `contract_statuses` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(50) NOT NULL UNIQUE,
    `name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `user_types` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(50) NOT NULL UNIQUE,
    `name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`id`)
);

CREATE TABLE `user_roles` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `code` VARCHAR(20) NOT NULL UNIQUE,
    `name` VARCHAR(50) NOT NULL,
    `description` TEXT NULL,
    `max_active_tasks` SMALLINT UNSIGNED NOT NULL DEFAULT 2,
    `max_in_progress_tasks` SMALLINT UNSIGNED NOT NULL DEFAULT 1,
    `max_active_requests` SMALLINT UNSIGNED NOT NULL DEFAULT 2,
    PRIMARY KEY (`id`)
);

-- ============================================
-- 2. ОСНОВНЫЕ ТАБЛИЦЫ (Гео, Пользователи, Объекты)
-- ============================================

CREATE TABLE `cities` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `name` VARCHAR(100) NOT NULL,
    `region` VARCHAR(100) NULL,
    `external_id` CHAR(36) NULL,
    PRIMARY KEY (`id`),
    INDEX `idx_cities_name` (`name`),
    INDEX `idx_cities_external_id` (`external_id`)
);

CREATE TABLE `streets` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `city_id` INT NOT NULL,
    `name` VARCHAR(150) NOT NULL,
    `street_type` VARCHAR(20) NULL,
    `external_id` CHAR(36) NULL,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`city_id`) REFERENCES `cities`(`id`) ON DELETE CASCADE,
    INDEX `idx_streets_external_id` (`external_id`)
);

CREATE TABLE `houses` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `street_id` INT NOT NULL,
    `house_number` VARCHAR(20) NOT NULL,
    `postal_code` VARCHAR(10) NULL,
    `external_id` CHAR(36) NULL,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`street_id`) REFERENCES `streets`(`id`) ON DELETE CASCADE,
    INDEX `idx_houses_external_id` (`external_id`)
);

CREATE TABLE `users` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `password` VARCHAR(128) NOT NULL,
    `last_login` DATETIME NULL,
    `is_superuser` BOOLEAN NOT NULL DEFAULT FALSE,
    `username` VARCHAR(50) NOT NULL UNIQUE,
    `email` VARCHAR(255) NOT NULL UNIQUE,
    `phone` VARCHAR(20) UNIQUE NULL,
    `user_type_ref_id` INT NOT NULL DEFAULT 1,
    `role_id` INT NULL,
    `is_active` BOOLEAN NOT NULL DEFAULT TRUE,
    `is_staff` BOOLEAN NOT NULL DEFAULT FALSE,
    `is_email_verified` BOOLEAN NOT NULL DEFAULT FALSE,
    `is_phone_verified` BOOLEAN NOT NULL DEFAULT FALSE,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`user_type_ref_id`) REFERENCES `user_types`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`role_id`) REFERENCES `user_roles`(`id`) ON DELETE SET NULL,
    INDEX `idx_users_username` (`username`),
    INDEX `idx_users_email` (`email`)
);

CREATE TABLE `employee_profiles` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `user_id` INT NOT NULL UNIQUE,
    `first_name` VARCHAR(50) NOT NULL,
    `last_name` VARCHAR(50) NOT NULL,
    `position` VARCHAR(100) NULL,
    `hire_date` DATE NULL,
    `internal_phone` VARCHAR(20) NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE
);

CREATE TABLE `client_profiles` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `user_id` INT NOT NULL UNIQUE,
    `first_name` VARCHAR(50) NOT NULL,
    `last_name` VARCHAR(50) NOT NULL,
    `middle_name` VARCHAR(50) NULL,
    `client_kind_ref_id` INT NOT NULL DEFAULT 1,
    `contact_method_id` INT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`client_kind_ref_id`) REFERENCES `client_kinds`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`contact_method_id`) REFERENCES `contact_methods`(`id`) ON DELETE SET NULL
);

CREATE TABLE `client_individual_details` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `profile_id` INT NOT NULL UNIQUE,
    `passport_series` VARCHAR(4) NULL,
    `passport_number` VARCHAR(6) NULL,
    `passport_issued_by` VARCHAR(255) NULL,
    `passport_issued_date` DATE NULL,
    `passport_code` VARCHAR(7) NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`profile_id`) REFERENCES `client_profiles`(`id`) ON DELETE CASCADE
);

CREATE TABLE `client_company_details` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `profile_id` INT NOT NULL UNIQUE,
    `company_inn` VARCHAR(10) NULL UNIQUE,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`profile_id`) REFERENCES `client_profiles`(`id`) ON DELETE CASCADE,
    INDEX `idx_company_inn` (`company_inn`)
);

CREATE TABLE `properties` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `title` VARCHAR(255) NULL,
    `operation_type_id` INT NOT NULL,
    `status_id` INT NOT NULL DEFAULT 1,
    `house_id` INT NOT NULL,
    `property_type_ref_id` INT NOT NULL DEFAULT 1,
    `owner_id` INT NULL,
    `coordinates_lat` DECIMAL(10,8) NULL,
    `coordinates_lon` DECIMAL(11,8) NULL,
    `price` DOUBLE NOT NULL,
    `area_total` DECIMAL(8,2) NULL,
    `rooms_count` INT NULL,
    `floor_number` INT NULL,
    `total_floors` INT NULL,
    `description` TEXT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`operation_type_id`) REFERENCES `operation_types`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`status_id`) REFERENCES `property_statuses`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`house_id`) REFERENCES `houses`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`property_type_ref_id`) REFERENCES `property_types`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`owner_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    INDEX `idx_properties_house` (`house_id`)
);

CREATE TABLE `property_external_sources` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `property_id` INT NOT NULL,
    `source_name` VARCHAR(50) NOT NULL,
    `external_id` VARCHAR(128) NOT NULL,
    `source_object_name` VARCHAR(255) NULL,
    `source_address` TEXT NULL,
    `source_rubric` VARCHAR(255) NULL,
    `synced_at` DATETIME NULL,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE CASCADE,
    INDEX `idx_property_source` (`property_id`, `source_name`)
);

CREATE TABLE `property_photos` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `property_id` INT NOT NULL,
    `image` VARCHAR(255) NULL,
    `url` TEXT NULL,
    `caption` VARCHAR(255) NULL,
    `is_cover` BOOLEAN NOT NULL DEFAULT FALSE,
    `is_hidden` BOOLEAN NOT NULL DEFAULT FALSE,
    `order` INT UNSIGNED NOT NULL DEFAULT 0,
    `uploaded_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE CASCADE,
    INDEX `idx_photos_property` (`property_id`)
);

CREATE TABLE `property_documents` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `property_id` INT NOT NULL,
    `document_name` VARCHAR(255) NOT NULL,
    `file_url` TEXT NOT NULL,
    `is_verified` BOOLEAN NOT NULL DEFAULT FALSE,
    `verified_by_id` INT NULL,
    `verified_at` DATETIME NULL,
    `uploaded_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`verified_by_id`) REFERENCES `users`(`id`) ON DELETE SET NULL
);

CREATE TABLE `property_status_history` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `property_id` INT NOT NULL,
    `old_status_id` INT NULL,
    `new_status_id` INT NOT NULL,
    `changed_by_id` INT NOT NULL,
    `changed_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`old_status_id`) REFERENCES `property_statuses`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`new_status_id`) REFERENCES `property_statuses`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`changed_by_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    INDEX `idx_history_property` (`property_id`)
);

CREATE TABLE `property_viewings` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `property_id` INT NOT NULL,
    `client_id` INT NOT NULL,
    `agent_id` INT NOT NULL,
    `scheduled_date` DATETIME NOT NULL,
    `notes` TEXT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`client_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`agent_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    INDEX `idx_viewings_scheduled` (`scheduled_date`)
);

-- ============================================
-- 3. ЗАЯВКИ, ЗАДАЧИ, СДЕЛКИ
-- ============================================

CREATE TABLE `requests` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `client_id` INT NOT NULL,
    `agent_id` INT NULL,
    `property_id` INT NULL,
    `operation_type_id` INT NOT NULL,
    `status_id` INT NOT NULL DEFAULT 1,
    `property_type` VARCHAR(50) NULL,
    `min_price` DOUBLE NULL,
    `max_price` DOUBLE NULL,
    `min_area` DECIMAL(8,2) NULL,
    `max_area` DECIMAL(8,2) NULL,
    `rooms_count` INT NULL,
    `address_preferences` TEXT NULL,
    `description` TEXT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    `closed_at` DATETIME NULL,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`client_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`agent_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`operation_type_id`) REFERENCES `operation_types`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`status_id`) REFERENCES `request_statuses`(`id`) ON DELETE RESTRICT,
    INDEX `idx_requests_client` (`client_id`)
);

CREATE TABLE `request_property_matches` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `request_id` INT NOT NULL,
    `property_id` INT NOT NULL,
    `agent_id` INT NOT NULL,
    `agent_note` TEXT NULL,
    `is_offered` BOOLEAN NOT NULL DEFAULT TRUE,
    `is_rejected` BOOLEAN NOT NULL DEFAULT FALSE,
    `is_confirmed` BOOLEAN NOT NULL DEFAULT FALSE,
    `confirmed_at` DATETIME NULL,
    `confirmed_by_id` INT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`request_id`) REFERENCES `requests`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`agent_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`confirmed_by_id`) REFERENCES `users`(`id`) ON DELETE SET NULL,
    INDEX `idx_match_request` (`request_id`),
    INDEX `idx_match_property` (`property_id`)
);

CREATE TABLE `deals` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `deal_number` VARCHAR(50) NOT NULL UNIQUE,
    `property_id` INT NOT NULL,
    `agent_id` INT NOT NULL,
    `client_id` INT NOT NULL,
    `operation_type_id` INT NOT NULL,
    `status_id` INT NULL,
    `request_id` INT NULL UNIQUE,
    `price_final` DOUBLE NOT NULL,
    `commission_percent` DECIMAL(5,2) NULL,
    `commission_amount` DOUBLE NULL,
    `deal_date` DATE NOT NULL,
    `notes` TEXT NULL,
    `contract_file` VARCHAR(255) NULL,
    `contract_status_ref_id` INT NOT NULL DEFAULT 1,
    `contract_error_message` TEXT NULL,
    `contract_requested_at` DATETIME NULL,
    `contract_processing_started_at` DATETIME NULL,
    `contract_generated_at` DATETIME NULL,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`agent_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`client_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`operation_type_id`) REFERENCES `operation_types`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`status_id`) REFERENCES `deal_statuses`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`request_id`) REFERENCES `requests`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`contract_status_ref_id`) REFERENCES `contract_statuses`(`id`) ON DELETE RESTRICT,
    INDEX `idx_deals_number` (`deal_number`),
    INDEX `idx_deals_client` (`client_id`)
);

CREATE TABLE `tasks` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `title` VARCHAR(255) NOT NULL,
    `description` TEXT NULL,
    `priority_ref_id` INT NOT NULL DEFAULT 2,
    `task_type_ref_id` INT NOT NULL DEFAULT 6,
    `status_id` INT NOT NULL,
    `assignee_id` INT NOT NULL,
    `created_by_id` INT NOT NULL,
    `client_id` INT NULL,
    `property_id` INT NULL,
    `request_id` INT NULL,
    `deal_id` INT NULL,
    `due_date` DATETIME NULL,
    `completed_at` DATETIME NULL,
    `result` TEXT NULL,
    `steps_log` JSON NOT NULL,
    `is_auto_closed` BOOLEAN NOT NULL DEFAULT FALSE,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`priority_ref_id`) REFERENCES `task_priorities`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`task_type_ref_id`) REFERENCES `task_types`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`status_id`) REFERENCES `task_statuses`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`assignee_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`created_by_id`) REFERENCES `users`(`id`) ON DELETE RESTRICT,
    FOREIGN KEY (`client_id`) REFERENCES `users`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`request_id`) REFERENCES `requests`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`deal_id`) REFERENCES `deals`(`id`) ON DELETE SET NULL,
    INDEX `idx_tasks_assignee` (`assignee_id`),
    INDEX `idx_tasks_status` (`status_id`)
);

-- 4. СЛУЖЕБНЫЕ ТАБЛИЦЫ (Email, Audit, Backup)
-- ============================================

CREATE TABLE `outgoing_emails` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `recipient_id` INT NOT NULL,
    `sender_id` INT NULL,
    `subject` VARCHAR(255) NOT NULL,
    `body` TEXT NOT NULL,
    `html_body` TEXT NOT NULL,
    `trigger_code` VARCHAR(64) NULL,
    `context` JSON NOT NULL,
    `status` VARCHAR(20) NOT NULL DEFAULT 'pending',
    `task_id` INT NULL,
    `request_id` INT NULL,
    `property_id` INT NULL,
    `error_message` TEXT NULL,
    `processing_started_at` DATETIME NULL,
    `sent_at` DATETIME NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`recipient_id`) REFERENCES `users`(`id`) ON DELETE CASCADE,
    FOREIGN KEY (`sender_id`) REFERENCES `users`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`task_id`) REFERENCES `tasks`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`request_id`) REFERENCES `requests`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE SET NULL,
    INDEX `idx_emails_trigger` (`trigger_code`),
    INDEX `idx_emails_status` (`status`)
);

CREATE TABLE `audit_logs` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `entity_type` VARCHAR(32) NOT NULL,
    `entity_id` INT UNSIGNED NOT NULL,
    `action_code` VARCHAR(64) NOT NULL,
    `action_label` VARCHAR(255) NOT NULL,
    `message` TEXT NOT NULL,
    `metadata` JSON NOT NULL,
    `actor_id` INT NULL,
    `property_id` INT NULL,
    `request_id` INT NULL,
    `task_id` INT NULL,
    `deal_id` INT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`actor_id`) REFERENCES `users`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`property_id`) REFERENCES `properties`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`request_id`) REFERENCES `requests`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`task_id`) REFERENCES `tasks`(`id`) ON DELETE SET NULL,
    FOREIGN KEY (`deal_id`) REFERENCES `deals`(`id`) ON DELETE SET NULL,
    INDEX `idx_audit_entity` (`entity_type`, `entity_id`),
    INDEX `idx_audit_action` (`action_code`),
    INDEX `idx_audit_created` (`created_at`)
);

CREATE TABLE `database_backups` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `filename` VARCHAR(255) NOT NULL,
    `file` VARCHAR(255) NOT NULL,
    `size_bytes` BIGINT UNSIGNED NOT NULL DEFAULT 0,
    `database_name` VARCHAR(255) NOT NULL,
    `engine_label` VARCHAR(120) NOT NULL,
    `tool_label` VARCHAR(120) NOT NULL,
    `created_by_id` INT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    FOREIGN KEY (`created_by_id`) REFERENCES `users`(`id`) ON DELETE SET NULL,
    INDEX `idx_backups_created` (`created_at`)
);
